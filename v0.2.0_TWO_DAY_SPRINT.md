### **Hour 3-7: HTTP REST Plugin** (5 hours)

**Create HttpRestWorkloadSimulator.java** (bare minimum):

```java
package com.cloudoptimizer.agent.simulator;

import com.cloudoptimizer.agent.model.RunResult;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.http.*;
import org.springframework.stereotype.Component;
import org.springframework.web.client.RestTemplate;
import java.util.concurrent.*;
import java.util.ArrayList;
import java.util.List;

@Component("http")
public class HttpRestWorkloadSimulator implements WorkloadSimulator {
    
    @Value("${http.base-url:http://localhost:8080}")
    private String baseUrl;
    
    @Value("${http.endpoint:/api/test}")
    private String endpoint;
    
    @Value("${http.method:GET}")
    private String method;
    
    private final RestTemplate restTemplate = new RestTemplate();
    
    @Override
    public RunResult executeLoad(int concurrency, int durationSeconds, double targetRps) {
        ExecutorService executor = Executors.newFixedThreadPool(concurrency);
        List<Future<Long>> futures = new ArrayList<>();
        
        long startTime = System.currentTimeMillis();
        long endTime = startTime + (durationSeconds * 1000L);
        
        // Submit requests
        while (System.currentTimeMillis() < endTime) {
            futures.add(executor.submit(() -> {
                long reqStart = System.nanoTime();
                try {
                    String url = baseUrl + endpoint;
                    if ("GET".equals(method)) {
                        restTemplate.getForEntity(url, String.class);
                    } else if ("POST".equals(method)) {
                        restTemplate.postForEntity(url, null, String.class);
                    }
                } catch (Exception e) {
                    // Log but continue
                }
                return System.nanoTime() - reqStart;
            }));
            
            // Rate limiting (simple)
            if (targetRps > 0) {
                try {
                    Thread.sleep((long)(1000.0 / targetRps));
                } catch (InterruptedException e) {
                    Thread.currentThread().interrupt();
                }
            }
        }
        
        executor.shutdown();
        try {
            executor.awaitTermination(10, TimeUnit.SECONDS);
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
        }
        
        // Collect results
        List<Long> latencies = new ArrayList<>();
        for (Future<Long> f : futures) {
            try {
                latencies.add(f.get());
            } catch (Exception e) {
                // Skip failed requests
            }
        }
        
        if (latencies.isEmpty()) {
            throw new RuntimeException("All requests failed!");
        }
        
        // Calculate metrics
        latencies.sort(Long::compareTo);
        double medianLatencyMs = latencies.get(latencies.size() / 2) / 1_000_000.0;
        double avgLatencyMs = latencies.stream().mapToLong(l -> l).average().orElse(0) / 1_000_000.0;
        double p95LatencyMs = latencies.get((int)(latencies.size() * 0.95)) / 1_000_000.0;
        
        long actualDuration = System.currentTimeMillis() - startTime;
        double rps = (latencies.size() * 1000.0) / actualDuration;
        
        return RunResult.builder()
            .concurrency(concurrency)
            .totalRequests(latencies.size())
            .medianLatencyMs(medianLatencyMs)
            .avgLatencyMs(avgLatencyMs)
            .p95LatencyMs(p95LatencyMs)
            .requestsPerSecond(rps)
            .costEstimateUsd(latencies.size() * 0.0001)  // Mock cost
            .build();
    }
    
    @Override
    public String getName() {
        return "http";
    }
}
```

```java
@Override
public boolean isHealthy() {
    try {
        String url = baseUrl + endpoint;
        ResponseEntity<String> response = restTemplate.getForEntity(url, String.class);
        return response.getStatusCode().is2xxSuccessful();
    } catch (Exception e) {
        return false;
    }
}
```

**Check health in RunnerMain:**
