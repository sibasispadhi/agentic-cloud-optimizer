### **1.2 Create WorkloadSimulator interface**

**File:** `src/main/java/com/cloudoptimizer/agent/simulator/WorkloadSimulator.java`

```java
package com.cloudoptimizer.agent.simulator;

import com.cloudoptimizer.agent.model.RunResult;

/**
 * Interface for workload simulators that execute load tests.
 * Implementations can test different types of services (HTTP, gRPC, DB, etc.)
 */
public interface WorkloadSimulator {
    
    /**
     * Execute a load test with the specified configuration.
     * 
     * @param config the load test configuration
     * @return results including latency, throughput, and cost metrics
     */
    RunResult executeLoad(LoadTestConfig config);
    
    /**
     * Get metadata about this simulator.
     * 
     * @return simulator metadata
     */
    SimulatorMetadata getMetadata();
    
    /**
     * Check if the simulator is healthy and ready to run tests.
     * 
     * @return true if healthy, false otherwise
     */
    boolean isHealthy();
    
    /**
     * Cleanup resources when simulator is no longer needed.
     */
    default void shutdown() {
        // Default: no-op, override if resources need cleanup
    }
}
```

```java
package com.cloudoptimizer.agent.simulator;

import java.util.HashMap;
import java.util.Map;

/**
 * Configuration for executing a load test.
 */
public class LoadTestConfig {
    private final int concurrency;
    private final int durationSeconds;
    private final double targetRps;
    private final Map<String, Object> customParams;
    
    private LoadTestConfig(Builder builder) {
        this.concurrency = builder.concurrency;
        this.durationSeconds = builder.durationSeconds;
        this.targetRps = builder.targetRps;
        this.customParams = builder.customParams;
    }
    
    public int getConcurrency() {
        return concurrency;
    }
    
    public int getDurationSeconds() {
        return durationSeconds;
    }
    
    public double getTargetRps() {
        return targetRps;
    }
    
    public Map<String, Object> getCustomParams() {
        return new HashMap<>(customParams);
    }
    
    public static Builder builder() {
        return new Builder();
    }
    
    public static class Builder {
        private int concurrency = 4;
        private int durationSeconds = 10;
        private double targetRps = 0.0;
        private Map<String, Object> customParams = new HashMap<>();
        
        public Builder concurrency(int concurrency) {
            this.concurrency = concurrency;
            return this;
        }
        
        public Builder durationSeconds(int durationSeconds) {
            this.durationSeconds = durationSeconds;
            return this;
        }
        
        public Builder targetRps(double targetRps) {
            this.targetRps = targetRps;
            return this;
        }
        
        public Builder customParam(String key, Object value) {
            this.customParams.put(key, value);
            return this;
        }
        
        public LoadTestConfig build() {
            return new LoadTestConfig(this);
        }
    }
}
```

## 📝 Step 2: Refactor Existing LoadRunner (1.5 hours)

### **2.1 Create DemoWorkloadSimulator**

**File:** `src/main/java/com/cloudoptimizer/agent/simulator/DemoWorkloadSimulator.java`

```java
package com.cloudoptimizer.agent.simulator;

import com.cloudoptimizer.agent.model.RunResult;
import com.cloudoptimizer.agent.service.LoadRunner;
import org.springframework.stereotype.Component;

/**
 * Demo workload simulator using the built-in /v1/work endpoint.
 * This is the same demo from v0.1.0, refactored to use the plugin interface.
 */
@Component("demo")
public class DemoWorkloadSimulator implements WorkloadSimulator {
    
    private final LoadRunner loadRunner;
    
    public DemoWorkloadSimulator(LoadRunner loadRunner) {
        this.loadRunner = loadRunner;
    }
    
    @Override
    public RunResult executeLoad(LoadTestConfig config) {
        // Delegate to existing LoadRunner
        return loadRunner.run(
            config.getConcurrency(),
            config.getDurationSeconds(),
            config.getTargetRps()
        );
    }
    
    @Override
    public SimulatorMetadata getMetadata() {
        return SimulatorMetadata.builder()
            .name("demo")
            .description("Built-in demo API simulator (v0.1.0 compatibility)")
            .version("0.2.0")
            .build();
    }
    
    @Override
    public boolean isHealthy() {
        // The demo endpoint is always available (runs in same process)
        return true;
    }
}
```

```java
package com.cloudoptimizer.agent.simulator;

import com.cloudoptimizer.agent.model.RunResult;
import com.cloudoptimizer.agent.service.LoadRunner;
import org.junit.jupiter.api.Test;
import org.junit.jupiter.api.extension.ExtendWith;
import org.mockito.InjectMocks;
import org.mockito.Mock;
import org.mockito.junit.jupiter.MockitoExtension;

import static org.junit.jupiter.api.Assertions.*;
import static org.mockito.ArgumentMatchers.*;
import static org.mockito.Mockito.when;

@ExtendWith(MockitoExtension.class)
class DemoWorkloadSimulatorTest {
    
    @Mock
    private LoadRunner loadRunner;
    
    @InjectMocks
    private DemoWorkloadSimulator simulator;
    
    @Test
    void testGetMetadata() {
        SimulatorMetadata metadata = simulator.getMetadata();
        
        assertNotNull(metadata);
        assertEquals("demo", metadata.getName());
        assertEquals("0.2.0", metadata.getVersion());
    }
    
    @Test
    void testIsHealthy() {
        assertTrue(simulator.isHealthy());
    }
    
    @Test
    void testExecuteLoad() {
        // Arrange
        LoadTestConfig config = LoadTestConfig.builder()
            .concurrency(4)
            .durationSeconds(10)
            .targetRps(0)
            .build();
        
        RunResult mockResult = RunResult.builder()
            .concurrency(4)
            .totalRequests(100)
            .medianLatencyMs(30.0)
            .build();
        
        when(loadRunner.run(anyInt(), anyInt(), anyDouble()))
            .thenReturn(mockResult);
        
        // Act
        RunResult result = simulator.executeLoad(config);
        
        // Assert
        assertNotNull(result);
        assertEquals(4, result.getConcurrency());
        assertEquals(100, result.getTotalRequests());
    }
}
```
